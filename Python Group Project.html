## 4) Using the train set, build four different models, using techniques listed below....

# Loading needed data and parameters:
from sklearn.linear_model import LogisticRegression
from sklearn.tree import DecisionTreeClassifier
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.model_selection import GridSearchCV


# Define models and their hyper-parameter spaces for grid search
models = {
    'Logistic Regression': LogisticRegression(max_iter=1000),
    'Decision Tree': DecisionTreeClassifier(),
    'Random Forest': RandomForestClassifier(),
    'Gradient Boosting': GradientBoostingClassifier()
}

param_grids = {
    'Decision Tree': {'max_depth': [3, 5, 10, None], 'min_samples_split': [2, 5, 10]},
    'Random Forest': {'n_estimators': [100, 200, 300], 'max_depth': [3, 5, 10, None], 'min_samples_split': [2, 5, 10]},
    'Gradient Boosting': {'n_estimators': [100, 200, 300], 'learning_rate': [0.01, 0.1, 0.5], 'max_depth': [3, 5, 10]}
}

# Initialize dictionaries to store the best parameters and scores
best_params = {}
best_scores = {}

# Perform fitting and hyper-parameter optimization
for name, model in models.items():
    if name == 'Logistic Regression':
        model.fit(X_train_scaled_df, y_train)
        best_scores[name] = model.score(X_train_scaled_df, y_train)
    else:
        grid_search = GridSearchCV(model, param_grids[name], cv=5, scoring='accuracy')
        grid_search.fit(X_train_scaled_df, y_train)
        models[name] = grid_search.best_estimator_
        best_params[name] = grid_search.best_params_
        best_scores[name] = grid_search.best_score_

# Print model performance summary
print("Model Performance Summary:\n")
for name in models.keys():
    print(f"{name} Model")
    if name in best_params:
        print(f"  Best Parameters: {best_params[name]}")
    print(f"  Best Training Accuracy Score: {best_scores[name]:.4f}\n")

## 6) Get the predicted probabilities from all four models for every record in the test set.
## Plot those probabilities on one chart to show the distribution of those probabilities for all four models. 
## Also, calculate the correlation matrix among all four predicted probabilities.

import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt

# Step 1: Get the Predicted Probabilities for the positive class from all four models:
probabilities = {}
for name, model in models.items():
    probabilities[name] = model.predict_proba(X_test_scaled_df)[:, 1]

# Step 2: Plot the distribution of these probabilities:
plt.figure(figsize=(10, 8))
colors = ['darkblue', 'orange', 'green', 'pink']
for (name, probs), color in zip(probabilities.items(), colors):
    sns.kdeplot(probs, fill=True, common_norm=False, alpha=.5, linewidth=0, label=name, color=color)

plt.title('Distribution of Predicted Probabilities')
plt.xlabel('Probability of Positive Class')
plt.ylabel('Density')
plt.legend()
plt.show()


# Step 3: Calculate the Correlation Matrix among all four predicted probabilities:
prob_df = pd.DataFrame(probabilities)
correlation_matrix = prob_df.corr()
print("Correlation Matrix among Predicted Probabilities:")
print(correlation_matrix)

# Visualizing the Correlation Matrix (for fun):
plt.figure(figsize=(8, 6))
sns.heatmap(correlation_matrix, annot=True, cmap='coolwarm', linewidths=.5)
plt.title('Correlation Matrix of Model Probabilities')
plt.show()
